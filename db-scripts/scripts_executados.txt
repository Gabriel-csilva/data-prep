
// --- TABELAS DE DOMÍNIO ---

Table etnia {
  id_etnia serial [primary key]
  descricao varchar(50) [not null]
}

Table categoria {
  id_categoria serial [primary key]
  nome varchar(100) [not null]
  descricao text
}

Table fornecedor {
  id_fornecedor serial [primary key]
  nome_fantasia varchar(150) [not null]
  cnpj varchar(18) [unique, not null]
  contato varchar(100)
}

Table promocao {
  id_promocao serial [primary key]
  nome varchar(100) [not null]
  desconto_percentual decimal(5,2) [not null]
  data_inicio date [not null]
  data_fim date [not null]
}

// --- COMPONENTES (CENÁRIO 5) ---

Table cores {
  id_cor serial [primary key]
  nome varchar(50) [not null]
  valor_adicional decimal(10,2) [default: 0.00]
  quantidade_estoque int [default: 0, not null]
}

Table materiais {
  id_material serial [primary key]
  nome varchar(50) [not null]
  valor_adicional decimal(10,2) [default: 0.00]
  quantidade_estoque int [default: 0, not null]
}

Table estampas {
  id_estampa serial [primary key]
  nome varchar(50) [not null]
  valor_adicional decimal(10,2) [default: 0.00]
  quantidade_estoque int [default: 0, not null]
}

Table textos_personalizados {
  id_texto serial [primary key]
  limite_caracteres int [not null]
  valor_adicional decimal(10,2) [default: 0.00]
}

// --- CLIENTES E ENDEREÇOS ---

Table cliente {
  id_cliente serial [primary key]
  id_etnia int [not null]
  nome varchar(150) [not null]
  cpf varchar(14) [unique, not null]
  email varchar(100) [unique, not null]
  idade int
}

Table endereco {
  id_endereco serial [primary key]
  id_cliente int [not null]
  tipo_endereco varchar(20) [not null, note: 'Entrega, Cobrança, Ambos']
  cep varchar(10) [not null]
  logradouro varchar(200) [not null]
  bairro varchar(100)
  cidade varchar(100) [not null]
  pais varchar(50) [default: 'Brasil']
}

// --- PRODUTOS E RELAÇÕES N:M ---

Table produto {
  id_produto serial [primary key]
  id_categoria int [not null]
  nome varchar(150) [not null]
  descricao text
  preco_base decimal(10,2) [not null]
}

Table produto_fornecedor {
  id_produto int [not null]
  id_fornecedor int [not null]
  Note: 'Chave Primária Composta'
  
  Indexes {
    (id_produto, id_fornecedor) [pk]
  }
}

Table categoria_promocao {
  id_categoria int [not null]
  id_promocao int [not null]
  
  Indexes {
    (id_categoria, id_promocao) [pk]
  }
}

// --- VENDAS E ITENS (CENÁRIO 2) ---

Table pedido {
  id_pedido serial [primary key]
  id_cliente int [not null]
  id_endereco_entrega int [not null]
  data_pedido timestamp [default: `now()`]
  status_pedido varchar(50) [default: 'Pendente']
  metodo_pagamento varchar(50)
  codigo_rastreio varchar(100)
  valor_total_pedido decimal(10,2) [not null]
}

Table item_pedido {
  id_item serial [primary key]
  id_pedido int [not null]
  id_produto int [not null]
  quantidade int [not null]
  preco_base_aplicado decimal(10,2) [not null]
  preco_personalizacoes_total decimal(10,2) [default: 0.00]
}

Table item_pedido_personalizacao {
  id_detalhe serial [primary key]
  id_item int [not null]
  tipo_opcao varchar(50) [not null]
  id_referencia int [not null]
  valor_cobrado decimal(10,2) [not null]
}

Table avaliacao {
  id_avaliacao serial [primary key]
  id_cliente int [not null]
  id_produto int [not null]
  nota int [not null, note: 'CHECK 1 a 5']
  comentario text
  data_avaliacao timestamp [default: `now()`]
}

// --- DEFINIÇÃO DOS RELACIONAMENTOS (FKs) ---

Ref: etnia.id_etnia < cliente.id_etnia
Ref: cliente.id_cliente < endereco.id_cliente
Ref: categoria.id_categoria < produto.id_categoria
Ref: produto.id_produto < produto_fornecedor.id_produto
Ref: fornecedor.id_fornecedor < produto_fornecedor.id_fornecedor
Ref: categoria.id_categoria < categoria_promocao.id_categoria
Ref: promocao.id_promocao < categoria_promocao.id_promocao
Ref: cliente.id_cliente < pedido.id_cliente
Ref: endereco.id_endereco < pedido.id_endereco_entrega
Ref: pedido.id_pedido < item_pedido.id_pedido
Ref: produto.id_produto < item_pedido.id_produto
Ref: item_pedido.id_item < item_pedido_personalizacao.id_item
Ref: cliente.id_cliente < avaliacao.id_cliente
Ref: produto.id_produto < avaliacao.id_produto
